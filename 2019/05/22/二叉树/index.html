二叉树的存储结构

1.链式存储的二叉树

代码实现

首先创建一个节点类

 package treeDemo1;
 public class TreeNode {
    //节点的权
    int value;
    //左子节点
    TreeNode LeftNode;
    //右子节点
    TreeNode RightNode;

    public TreeNode(int value){
        this.value=value;
    }

    //设置左子节点
    public void setLeftNode(TreeNode leftNode) {
        this.LeftNode = leftNode;
    }

    //设置右子节点
    public void setRightNode(TreeNode rightNode) {
        RightNode = rightNode;
    }

    //前序遍历
    public void frontshow() {
        //先遍历当前节点的内容
        System.out.println(value);
        //遍历左节点
        if(LeftNode!=null){
            LeftNode.frontshow();
        }
        //遍历右节点
        if(RightNode!=null){
            RightNode.frontshow();
        }

    }

    //中序遍历树
    public void middleshow() {
        //遍历左节点
        if(LeftNode!=null){
            LeftNode.middleshow();
        }

        //遍历当前节点
        System.out.println(value);

        //遍历右节点
        if(RightNode!=null){
            RightNode.middleshow();
        }

    }

    //后序遍历树
    public void aftershow() {
        //遍历左节点
        if(LeftNode!=null){
            LeftNode.aftershow();
        }

        //遍历右节点
        if(RightNode!=null){
            RightNode.aftershow();
        }

        //遍历当前节点
        System.out.println(value);

    }

    //前序查找
    public TreeNode frontSearch(int i) {
        TreeNode target=null;
        //对比当前节点的值
        if(this.value==i){
            return this;
        //当前查找的值不是当前的节点
        }else {
            //查找左子节点
            if(LeftNode!=null){
            //有可能查到，也可以查不到，如果查不到的话，target还是一个null
                target=LeftNode.frontSearch(i);
            }
            //如果不为空，说明在左子节点中已经找到
            if(target!=null){
                return target;
            }
            //查找右子节点
            if(RightNode!=null){
                target=RightNode.frontSearch(i);
            }
        }
        return target;

    }

    //中序查找
    public TreeNode middleSearch(int i) {
        TreeNode target=null;
        //查找左子节点
        if(LeftNode!=null){
        //有可能查到，也可以查不到，如果查不到的话，target还是一个null
            target=LeftNode.middleSearch(i);
        }
        //如果不为空，说明在左子节点中已经找到
        if(target!=null){
            return target;
        }
        //查找当前节点
        else if(this.value==i){
            return this;
        }
        //查找右子节点
        else if(RightNode!=null){
            target=RightNode.middleSearch(i);
        }
        return target;
    }

    //后序查找
    public TreeNode afterSearch(int i) {
        TreeNode target=null;
        //查找左子节点
        if(LeftNode!=null){
        //有可能查到，也可以查不到，如果查不到的话，target还是一个null
            target=LeftNode.afterSearch(i);
        }
        //如果不为空，说明在左子节点中已经找到
        if (target != null) {
            return target;
        }
        //查找右子节点
        else if(RightNode!=null){
            target=RightNode.afterSearch(i);
        }
        //查找当前节点
        else if(this.value==i){
            return this;
        }
        return target;
    }

    //删除一个子树
    public void delete(int i) {
        TreeNode parent=this;
        //判断左子节点
        if(parent.LeftNode!=null&&parent.LeftNode.value==i){
            parent.LeftNode=null;
            return;
        }
        //判断右子节点
        if(parent.RightNode!=null&&parent.RightNode.value==i){
            parent.RightNode=null;
            return;
        }
        //递归检查并删除左子节点
        parent=LeftNode;
        if(parent!=null){
            parent.delete(i);
        }
        //递归检查并删除右子节点
        parent=RightNode;
        if(parent!=null){
            parent.delete(i);
        }

    }
}

再创建一个二叉树类

 package treeDemo1;
 public class BinaryTree {
     //根节点
     TreeNode rootNode;

     //设置根节点
     public void setRootNode(TreeNode rootNode) {
        this.rootNode = rootNode;
    }

     //获取根节点
     public TreeNode getRootNode() {
        return rootNode;
    }

    //前序遍历树
    public void frontshow() {
        if(rootNode!=null){
            rootNode.frontshow();
        }
    }

    //中序遍历树
    public void middleshow() {
        if(rootNode!=null){
            rootNode.middleshow();
        }
    }

    //后序遍历树
    public void aftershow() {
        if(rootNode!=null){
            rootNode.aftershow();
        }   
    }

    //前序查找
    public TreeNode frontSearch(int i) {
        return rootNode.frontSearch(i);
    }

    //中序查找
    public TreeNode middleSearch(int i) {
        return rootNode.middleSearch(i);
    }

    //后序查找
    public TreeNode afterSearch(int i) {
        return rootNode.afterSearch(i);
    }

    //删除一个子树
    public void delete(int i) {
        if(rootNode.value==i){
            rootNode=null;
        }else{
            rootNode.delete(i);
        }

    }

}

测试代码

 package treeDemo1;
 public class TestBinaryTree {
    public static void main(String[] args) {
        //创建一棵树
        BinaryTree binTree=new BinaryTree();
        //创建一个根节点
        TreeNode rootNode=new TreeNode(1);
        //将根节点赋给树
        binTree.setRootNode(rootNode);
        //创建一个左节点
        TreeNode rootL=new TreeNode(2);
        //把新创建的节点设置为根节点的左子节点
        rootNode.setLeftNode(rootL);
        //创建一个右子节点
        TreeNode rootR=new TreeNode(3);
        //把新创建的子节点设置为根节点的右子节点
        rootNode.setRightNode(rootR);
        //为第二层的左节点创建两个子节点
        rootL.setLeftNode(new TreeNode(4));
        rootL.setRightNode(new TreeNode(5));
        //为第二层的右节点创建两个子节点
        rootR.setLeftNode(new TreeNode(6));
        rootR.setRightNode(new TreeNode(7));
        //前序遍历树
        binTree.frontshow();
        System.out.println("----------------------------");
        //中序遍历树
        binTree.middleshow();
        System.out.println("----------------------------");
        //后序遍历树
        binTree.aftershow();
        System.out.println("----------------------------");
        //前序查找
        TreeNode result=binTree.frontSearch(5);
        System.out.println(result);
        //中序查找
        TreeNode result1=binTree.middleSearch(5);
        System.out.println(result1);
        //后序查找
        TreeNode result2=binTree.afterSearch(5);
        System.out.println(result2);
        System.out.println("----------------------------");
        //删除一个子树
        binTree.delete(2);
        binTree.frontshow();
    }
}

发表于 2019-08-09 | 更新于 2019-08-15 | 分类于 数据库

use test;

/*
     dept：部门表

  deptno：     部门编号；  
  dname：      部门名称； 
  loc：        地理位置；
*/

create table dept (
  deptno int (2) not null,
  dname varchar (14),
  loc varchar (13) 
);

-- 添加主键
-- alter table 表 add primary key (主键字段名);
alter table dept add primary key (deptno);

/*
  emp：员工表

  empno：       员工编号；  
  ename：       员工名字；  
  job：         职位； 
  mgr:          上司； 
  hiredate：    入职时间；
  sal：         基本工资；  
  comm：        补贴；      
  deptno：      所属部门编号；
*/

  create table emp(
      empno int(4) not null,
      ename varchar(10),
      job varchar(9),
      mgr int(4),
      hiredate date,
      sal int(7 ),
      comm int(7 ),
      deptno int(2)
  )

  -- 添加主键
  -- alter table 表 add primary key (主键字段名);
  alter table emp add primary key (empno);
  desc emp;

  -- 添加外键约束
  /*
  alter table 从表 add constraint 自己建的外键名 
  foreign key (从表里的外键字段) 
  references 主表(映射主表中的主键字段);
  */
  alter table emp add constraint f_ed_key
  foreign key (deptno) 
  references dept(deptno)
